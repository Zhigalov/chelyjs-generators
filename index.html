Generator

ES6 ES2015

A generator can pause itself in mid-execution, and can be resumed either right away or at a later time.

Generator

  • node >= 0.11 --harmony или --harmony-generators
  • TODO: поддержка браузеров

Generator. Declaration

        function *foo() {
            // ...
        }

Generator. Execution

        function *foo(name) {
            console.log('Hello, ' + name);
        }
         
        foo('EkbJS');

Console:

        >

Generator. Execution

        function *foo(name) {
            console.log('Hello, ' + name);
        }
         
        var it = foo('EkbJS');
        it.next();

Console:

        > Hello, EkbJS

Generator. yield

        function *foo(name) {
            console.log('Hello, ' + name);
            yield;
            console.log('Goodbye, ' + name);
        }
         
        var it = foo('EkbJS');
        it.next();
        // Do somethings...
        it.next();

Console:

            >
            > Hello, EkbJS
            >
            > Goodbye, EkbJS

Generator. Receive data

        function *foo() {
            yield 'Some data';
        }
         
        var it = foo();
        console.log(it.next());

Console:

            > {value: "Some data", done: false}

Generator. Send data

        function *foo() {
            var name = yield;
            console.log('Hello, ' + name);
        }
         
        var it = foo();
        it.next();
        it.next('EkbJS');

Console:

            > Hello, EkbJS

Generator. Send data

            var name = 'EkbJS';
            console.log('Hello, ' + name);
        }
         
        var it = foo();
        it.next();
        it.next('EkbJS');

Console:

            > Hello, EkbJS

Generator. yield *

        function *foo() {
            yield 2;
            yield 3;
        }
         
        function *bar() {
            yield 1;
            yield *foo();
        }
         
        var it = bar();
        console.log(it.next());
        console.log(it.next());
        console.log(it.next());

Console:

            > {value: 1, done: false}
            > {value: 2, done: false}
            > {value: 3, done: false}

Generator. yield *

        function *foo() {
            yield 2;
            yield 3;
        }
         
        function *bar() {
            yield 1;
            yield *foo();
        }
         
        for(var val of bar()) {
            console.log(val);
        }

Console:

            > 1
            > 2
            > 3
